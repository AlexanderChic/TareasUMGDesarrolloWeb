void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        for (int j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                // Intercambiar elementos
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr)/sizeof(arr[0]);
    
    cout << "Array original: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    
    bubbleSort(arr, n);
    
    cout << "\nArray ordenado: ";

        cout << arr[i] << " ";
